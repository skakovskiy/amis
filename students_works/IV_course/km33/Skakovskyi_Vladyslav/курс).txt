  CREATE OR REPLACE FORCE EDITIONABLE VIEW "C##JAKE"."ACT" ("USER_NAME", "USER_LOGIN") AS

  CREATE OR REPLACE FORCE EDITIONABLE VIEW "C##JAKE"."AC2" ("RESUALT", "USERS_LOGIN_FK", "SUBJECT_NAME_FK") AS 
  (
select resualt  , users_login_fk , subject_name_fk from LISTS inner join comission on LISTS.comission_fk = comission_id inner join list_teacher on COMISSION_ID =  LIST_TEACHER.COMISSION_FK );

  CREATE OR REPLACE FORCE EDITIONABLE VIEW "C##JAKE"."B4" ("COMISSION_FK", "TEACHER", "SUBJECT_NAME_FK", "DATE_COMISSION", "TIME_COMISSION", "USERS_LOGIN_FK") AS 
  (
 Select lists.COMISSION_FK,user_name  AS TEACHER , subject_name_fk , DATE_COMISSION , TIME_COMISSION ,USERS_LOGIN_FK   FROM  USERs  inner join  LIST_TEACHER on USERS.USER_LOGIN= LIST_TEACHER.USER_LOGIN_FK  
inner join   Comission on LIST_TEACHER.COMISSION_FK= COMISSION.COMISSION_ID inner join LISTS ON COMISSION.COMISSION_ID=LISTS.COMISSION_FK );

  CREATE OR REPLACE FORCE EDITIONABLE VIEW "C##JAKE"."LIST_FOR_ALL1" ("COMISSION_FK", "TEACHER", "SUBJECT_NAME_FK", "DATE_COMISSION", "TIME_COMISSION", "USER_NAME") AS 
  (

  CREATE OR REPLACE FORCE EDITIONABLE VIEW "C##JAKE"."LIST_FOR_TEACHER1" ("COMISSION_FK", "USER_LOGIN", "TEACHER", "SUBJECT_NAME_FK", "DATE_COMISSION", "TIME_COMISSION", "USER_NAME") AS

  CREATE OR REPLACE FORCE EDITIONABLE VIEW "C##JAKE"."LIST_LISTS" ("NUMBER_LIST", "NUMBER_STUDENT", "STUDENT_NAME", "USER_LOGIN", "NAME_GROUP", "NUMBER_GROUP", "RESUALT", "DATE_COMISSION", "TIME_COMISSION") AS 
  (
Select NUMBER_LIST , NUMBER_STUDENT , Users.user_name as student_name ,users.user_login , NAME_GROUP , NUMBER_GROUP, RESUALT , DATE_COMISSION, TIME_COMISSION  From USERS inner join LISTS on users.user_login=LISTS.USERS_LOGIN_FK inner join COMISSION
